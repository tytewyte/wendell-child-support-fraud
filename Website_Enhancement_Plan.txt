// AI Agent System
class AIAgent {
    constructor() {
        this.apiKey = 'YOUR_OPENAI_API_KEY'; // Store securely
        this.conversationHistory = this.loadHistory();
        this.caseInfo = this.loadCaseInfo();
        this.init();
    }

    init() {
        this.setupEventListeners();
        this.loadMessages();
        this.checkMembership();
    }

    checkMembership() {
        const isMember = localStorage.getItem('wendell_member') === 'true';
        if (!isMember) {
            window.location.href = 'index.html#membership';
            alert('AI Agent is a Pro member feature. Please subscribe to access.');
        }
    }

    setupEventListeners() {
        // Send message
        document.getElementById('send-message').addEventListener('click', () => this.sendMessage());
        document.getElementById('chat-input').addEventListener('keypress', (e) => {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                this.sendMessage();
            }
        });

        // Quick actions
        document.querySelectorAll('.action-btn').forEach(btn => {
            btn.addEventListener('click', () => this.handleQuickAction(btn.dataset.action));
        });

        // Case info form
        document.getElementById('case-info-form').addEventListener('submit', (e) => {
            e.preventDefault();
            this.saveCaseInfo();
        });

        // Clear chat
        document.getElementById('clear-chat').addEventListener('click', () => this.clearHistory());

        // File upload
        const uploadArea = document.getElementById('upload-area');
        const fileInput = document.getElementById('file-input');

        uploadArea.addEventListener('click', () => fileInput.click());
        uploadArea.addEventListener('dragover', (e) => {
            e.preventDefault();
            uploadArea.classList.add('drag-over');
        });
        uploadArea.addEventListener('dragleave', () => {
            uploadArea.classList.remove('drag-over');
        });
        uploadArea.addEventListener('drop', (e) => {
            e.preventDefault();
            uploadArea.classList.remove('drag-over');
            this.handleFiles(e.dataTransfer.files);
        });
        fileInput.addEventListener('change', (e) => {
            this.handleFiles(e.target.files);
        });
    }

    async sendMessage() {
        const input = document.getElementById('chat-input');
        const message = input.value.trim();
        
        if (!message) return;

        // Add user message to UI
        this.addMessage(message, 'user');
        input.value = '';

        // Show typing indicator
        this.showTypingIndicator();

        try {
            // Call OpenAI API
            const response = await this.callOpenAI(message);
            
            // Remove typing indicator
            this.hideTypingIndicator();
            
            // Add AI response to UI
            this.addMessage(response, 'agent');
            
            // Save conversation
            this.saveHistory();
        } catch (error) {
            console.error('AI Agent error:', error);
            this.hideTypingIndicator();
            this.addMessage('Sorry, I encountered an error. Please try again.', 'agent');
        }
    }

    async callOpenAI(userMessage) {
        const systemPrompt = `You are a legal research assistant helping fathers fight child support fraud. You have access to:
- Complete book 'Fighting Back' by Charles Wendell Avant Jr.
- State-specific motion templates for all 50 states
- Legal research on fraud, res judicata, and due process
- Document analysis capabilities

The user's case information:
State: ${this.caseInfo.state || 'Not specified'}
Case Type: ${this.caseInfo.caseType || 'Not specified'}

Your role:
- Interview users about their case
- Analyze evidence for contradictions
- Draft motions and affidavits
- Provide step-by-step guidance
- Simulate opposition for practice

Always:
- Be encouraging and supportive
- Cite specific statutes and case law
- Provide actionable next steps
- Remind users to verify information
- Emphasize you're a research partner, not a lawyer`;

        const response = await fetch('https://api.openai.com/v1/chat/completions', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${this.apiKey}`
            },
            body: JSON.stringify({
                model: 'gpt-4',
                messages: [
                    { role: 'system', content: systemPrompt },
                    ...this.conversationHistory,
                    { role: 'user', content: userMessage }
                ],
                temperature: 0.7,
                max_tokens: 1000
            })
        });

        const data = await response.json();
        const aiMessage = data.choices[0].message.content;

        // Add to conversation history
        this.conversationHistory.push(
            { role: 'user', content: userMessage },
            { role: 'assistant', content: aiMessage }
        );

        return aiMessage;
    }

    addMessage(content, type) {
        const messagesContainer = document.getElementById('chat-messages');
        const messageDiv = document.createElement('div');
        messageDiv.className = `message ${type}-message`;
        
        const avatar = document.createElement('div');
        avatar.className = 'message-avatar';
        avatar.textContent = type === 'user' ? 'ğŸ‘¤' : 'ğŸ¤–';
        
        const contentDiv = document.createElement('div');
        contentDiv.className = 'message-content';
        contentDiv.innerHTML = this.formatMessage(content);
        
        messageDiv.appendChild(avatar);
        messageDiv.appendChild(contentDiv);
        messagesContainer.appendChild(messageDiv);
        
        // Scroll to bottom
        messagesContainer.scrollTop = messagesContainer.scrollHeight;
    }

    formatMessage(content) {
        // Convert markdown-style formatting to HTML
        return content
            .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
            .replace(/\*(.*?)\*/g, '<em>$1</em>')
            .replace(/\n/g, '<br>')
            .replace(/```(.*?)```/gs, '<pre><code>$1</code></pre>');
    }

    showTypingIndicator() {
        const indicator = document.createElement('div');
        indicator.id = 'typing-indicator';
        indicator.className = 'message agent-message';
        indicator.innerHTML = `
            <div class="message-avatar">ğŸ¤–</div>
            <div class="message-content">
                <div class="typing-dots">
                    <span></span><span></span><span></span>
                </div>
            </div>
        `;
        document.getElementById('chat-messages').appendChild(indicator);
    }

    hideTypingIndicator() {
        const indicator = document.getElementById('typing-indicator');
        if (indicator) indicator.remove();
    }

    handleQuickAction(action) {
        const prompts = {
            'analyze-evidence': 'I need help analyzing evidence for contradictions. I have multiple audits/affidavits that don\'t match.',
            'draft-motion': 'I need help drafting a Motion to Vacate Judgment based on fraud.',
            'research-statute': 'Can you explain the relevant statutes for my state regarding fraud on the court?',
            'prepare-hearing': 'I need help preparing for my upcoming hearing. Can you simulate opposition arguments?'
        };

        document.getElementById('chat-input').value = prompts[action];
        this.sendMessage();
    }

    saveCaseInfo() {
        const state = document.getElementById('state-select').value;
        const caseType = document.getElementById('case-type').value;
        
        this.caseInfo = { state, caseType };
        localStorage.setItem('wendell_case_info', JSON.stringify(this.caseInfo));
        
        this.showNotification('Case information saved!', 'success');
    }

    loadCaseInfo() {
        const saved = localStorage.getItem('wendell_case_info');
        return saved ? JSON.parse(saved) : {};
    }

    saveHistory() {
        localStorage.setItem('wendell_chat_history', JSON.stringify(this.conversationHistory));
    }

    loadHistory() {
        const saved = localStorage.getItem('wendell_chat_history');
        return saved ? JSON.parse(saved) : [];
    }

    loadMessages() {
        this.conversationHistory.forEach((msg, index) => {
            if (index % 2 === 0) {
                this.addMessage(msg.content, 'user');
            } else {
                this.addMessage(msg.content, 'agent');
            }
        });
    }

    clearHistory() {
        if (confirm('Are you sure you want to clear your chat history?')) {
            this.conversationHistory = [];
